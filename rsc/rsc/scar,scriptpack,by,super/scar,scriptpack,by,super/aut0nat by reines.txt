Procedure gettime;
Begin
 DecodeTime(Now,Hour,Min,Sec,MSec);
 nMin:=inttostr(Min);
 nHour:=inttostr(Hour);

 If(Hour<10)Then nHour:='0'+inttostr(Hour);
End;
