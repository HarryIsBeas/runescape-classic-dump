procedure sendkeyshuman(str:string);
var
  x1 : integer;
begin
  x1:=1
  repeat
    wait(random(200)+10);
    sendkeyssilent(copy(str,x1,1));
    x1 := x1+1;
  until x1 > length(str)
end;

procedure TalkToRand;
begin
  repeat
  if (FindColorSpiral(a,b,16711680,5,345,510,430))or
  (FindColorSpiral(a,b,16777215,5,345,510,430))then
      begin
        TheStatus:='Talking To Random Event'
        Status(TheStatus)
        Clickmouse(a,b,true);
        wait(2000+random(1000));
        randomtalked:=true
      end;
  until(GetColor(20,434)=0)
end;

procedure FindFlag;
var
  m : integer;
begin
  Wait(200);
  M:=0;
  if(FindBitmapIn(RunescapeFlag,a,b,570,5,725,162))then
    begin
      repeat
        wait(500);
        M:=M+1;
      until(FindBitmapIn(RunescapeFlag,a,b,633,64,665,90))and
      (not(GetColor(200,148)=16777215))or
      (not FindBitmapIn(RunescapeFlag,a,b,570,5,725,160))and(M>9);
       Wait(100);
    end;
end;

procedure FindFight;
var
  leaveloop, placecheck : integer;
begin
  if(Findcolorspiral(a,b,65280,180,90,320,230))then wait(1000);
  if(FindColorSpiral(a,b,65280,180,90,320,230))then
    begin
      SetRunon;
      wait(random(10)+50)
      if(Direction='NE')or(direction='')then
        clickmouse(680,35,true)
      if(direction='N')then
        clickmouse(660,20,true)
      if(direction='NW')then
        clickmouse(620,25,true)
      if(direction='W')then
        clickmouse(600,70,true)
      if(direction='SW')then
        clickmouse(620,130,true)
      if(direction='S')then
        clickmouse(645,145,true)
      if(direction='SE')then
        clickmouse(688,124,true)
      if(direction='E')then
        clickmouse(703,93,true)
      wait(100)
      findFlag;
      wait(random(1000)+10000)
      ClickMouse(615,110,true);
      Wait(100);
      findFlag;
      SetRunOff;
      a:=random(5)+675
      b:=random(5)+45
      MoveMouseSmoothEx(a,b,1,6,15,20,20);
      wait(random(500)+500)
      clickmouse(a,b,true)
      leaveloop:=0
      fightfound:=true
      repeat
        placecheck:= getcolor(27,36)
        wait(100)
        if(getcolor(27,36) = placecheck)then
          Leaveloop:=1
      until(Leaveloop=1)
    end;
end;


procedure FindText;
var
  counter : integer;
begin
  if(findcolor(a,b,65535,0,0,515,338)) then
    wait(200)
  if(findcolor(a,b,65535,0,0,515,338))then
    begin
      repeat
        counter:=counter+2+random(1)
        movemousesmooth(a+counter,b+15+random(3))
        wait(5)
        if(findbitmaptolerancein(Talk,a,b,01,1,500,500,100))or(findbitmaptolerancein(TalkY,a,b,01,1,500,500,100))then
          begin
            getmousepos(a,b);
            clickmouse(a,b,false)
            findflag;
            wait(random(100)+200)
            if(findbitmap(Talk,c,d))or(FindBitmap(TalkY,c,d))then
              begin
                clickmouse(c,d,true)
                TalkToRand;
              end;
          end;
      until(counter>=250);
    end;
end;

procedure RandomSong;
var ab : integer;
begin
  if(getcolor(730,495)<>2829182)then
    OpenMusic;
  wait(random(300)+300)
  ab:=random(2)
  if(ab<2)then
    begin
      a:=random(5)+725
      b:=random(5)+265
      MoveMouseSmoothEx(a,b,1,6,15,20,20);
      wait(random(300)+300)
      holdmouse(a,b,true)
      wait(random(400)+200)
      releasemouse(a,b,true)
    end;
  if(ab=2)then
    begin
      a:=random(5)+725
      b:=random(5)+435
      MoveMouseSmoothEx(a,b,1,6,15,20,20);
      wait(random(300)+300)
      holdmouse(a,b,true)
      wait(random(400)+200)
      releasemouse(a,b,true)
    end;
  ab:=0
  repeat
    ab:=ab+1
    a:=random(40)+570
    b:=random(5)+325
    wait(100)
  until(getcolor(a,b)=65280)or(ab=50)
  MoveMouseSmoothEx(a,b,1,6,15,20,20);
  wait(random(300)+300)
  clickmouse(a,b,true)
